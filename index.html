var note=(document.getElementById(“noNote”)||{}).value||newOrd.note||””;

var items=gi();

if(editId){

// Update existing order — keep original prices for existing items, snapshot new ones

var orders=go();

var orig=orders.find(function(x){return x.id===editId;})||{};

var prices=Object.assign({},orig.prices||{});

Object.keys(newOrd.items).forEach(function(it){

if(!prices[it])prices[it]=items[it]?items[it].price:0;

});

var updated=orders.map(function(o){

return o.id===editId?Object.assign({},o,{customerName:name,date:newOrd.date,items:Object.assign({},newOrd.items),prices:prices,note:note}):o;

});

so(updated);

editId=null;newOrd={cname:””,date:tday(),items:{},note:””};

curOV=“list”;showToast(“Order updated ✓”);renderOrders();updateHeader();

} else {

// New order — snapshot current prices
